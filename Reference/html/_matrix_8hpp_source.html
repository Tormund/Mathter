396 template <class T, int Dim, eMatrixOrder Order, eMatrixLayout Layout, bool Packed>
397 class MatrixProjectiveBase {
398 protected:
399  using MatrixT = Matrix<T, Dim, Dim, Order, Layout, Packed>;
400 public:
401  static MatrixT Perspective(T fovX, const Vector<T, Dim-2, Packed>& ratios, T nearPlane, T farPlane, T projNearPlane = 0, T projFarPlane = 1) {
402  assert((nearPlane < 0 && farPlane < nearPlane) || (0 < nearPlane && nearPlane < farPlane));
403 
404  MatrixT m;
405  m.SetZero();
406  // Layout be like (precede_vector):
407  // w 0 0 0
408  // 0 h 0 0
409  // 0 0 A B
410  // 0 0 C 0
411  fovX = abs(fovX);
412  T N = nearPlane;
413  T F = farPlane;
414  T n = projNearPlane;
415  T f = projFarPlane;
416  T C = nearPlane < T(0) ? T(-1) : T(1);
417  T A = C*(f*F - n*N) / (F - N);
418  T B = C*F*N*(n - f) / (F - N);
419  Vector<T, Dim-2, Packed> adjRatios = ratios(0) / ratios;
420  T w = tan(T(0.5)*fovX);
421  adjRatios /= w;
422  for (int i = 0; i < adjRatios.Dimension(); ++i) {
423  m(i, i) = adjRatios(i);
424  }
425  m(m.RowCount() - 2, m.ColumnCount() - 2) = A;
426  if (Order == eMatrixOrder::FOLLOW_VECTOR) {
427  std::swap(B, C);
428  }
429  m(m.RowCount() - 2, m.ColumnCount() - 1) = B;
430  m(m.RowCount() - 1, m.ColumnCount() - 2) = C;
431  return m;
432  }
433 
434  MatrixT& SetPerspective(T fovX, const Vector<T, Dim - 2, Packed>& ratios, T nearPlane, T farPlane, T projNearPlane = 0, T projFarPlane = 1) {
435  self() = Perspective(fovX, ratios, nearPlane, farPlane, projNearPlane, projFarPlane);

572  VectorT columns[SpaceDim];
573  std::array<const VectorT*, SpaceDim - 1> crossTable = {};
574  for (int i = 0; i < (int)bases.size(); ++i) {
575  crossTable[i] = &bases[i];
576  }
577  crossTable.back() = &columns[SpaceDim - 1];
578  auto elem = [&matrix](int i, int j) -> T& {
579  return Order == eMatrixOrder::FOLLOW_VECTOR ? matrix(i, j) : matrix(j, i);
580  };
581 
582  // calculate columns of the rotation matrix
583  int j = SpaceDim - 1;
584  columns[j] = Normalized(eye - target); // right-handed: camera look towards -Z
585  do {
586  --j;
587 
588  columns[SpaceDim - j - 2] = Normalized(Cross(crossTable));
589 
590  // shift bases
591  for (int s = 0; s < j; ++s) {
592  crossTable[s] = crossTable[s + 1];
593  }
594  crossTable[j] = &columns[SpaceDim - j - 2];
595  } while (j > 0);
596 
597  // flip columns
598  for (int i = 0; i < SpaceDim; ++i) {
599  if (flipAxes[i]) {
600  columns[i] *= -1;
601  }
602  }
603 
604  // copy columns to matrix
605  for (int i = 0; i < SpaceDim; ++i) {
606  for (int j = 0; j < SpaceDim; ++j) {
607  elem(i, j) = columns[j][i];
608  }
609  }
610 
611  // calculate translation of the matrix
612  for (int j = 0; j < SpaceDim; ++j) {
613  elem(SpaceDim, j) = -Dot(eye, columns[j]);
614  }
615 
616  // clear additional elements
617  constexpr int AuxDim = Rows < Columns ? Rows : Columns;
618  if (AuxDim > SpaceDim) {
619  for (int i = 0; i < SpaceDim; ++i) {
620  elem(i, AuxDim - 1) = 0;
621  }
622  elem(SpaceDim, AuxDim - 1) = 1;
623  }
624 
625  return matrix;
626  }
627 
628  // Specialization for 2D
629  template <class =std::enable_if<SpaceDim == 2, int>::type>
630  static MatrixT LookAt(const VectorT& eye, const VectorT& target, bool positiveYForward = true, bool flipX = false) {
631  return LookAt(eye, target, {}, { flipX, positiveYForward });
632  }

637  return LookAt(eye, target, { up }, { flipX, flipY, positiveZForward });

1497  }
1498

1503  for (int k = 0; k <= j - 1; ++k) {
1504  Lij -= L(i, k)*U(k, j);

1515  U(j, k) = Ujk;
1516  }
