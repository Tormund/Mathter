242 // Specialization for floats.
243 template <class T>
244 bool AlmostEqual(T d1, T d2, std::true_type) {
245  if (d1 < 1e-38 && d2 < 1e-38) {
246  return true;
247  }
248  if (d1 == 0 && d2 < 1e-4 || d2 == 0 && d1 < 1e-4) {
249  return true;
250  }
251  T scaler = pow(T(10), floor(log10(abs(d1))));
252  d1 /= scaler;
253  d2 /= scaler;
254  d1 *= T(1000.0);
255  d2 *= T(1000.0);
256  return round(d1) == round(d2);
257 }

1335 template <class T, int Dim, bool Packed>
1336 auto VectorSpecialOps<T, Dim, Packed>::Cross(const std::array<const VectorT*, Dim - 1>& args) -> VectorT {
1337  VectorT result;
1338  Matrix<T, Dim - 1, Dim - 1> detCalc;
1339 
1340  // Calculate elements of result on-by-one
1341  int sign = 2 * (Dim % 2) - 1;
1342  for (int base = 0; base < result.Dimension(); ++base, sign *= -1) {
1343  // Fill up sub-matrix the determinant of which yields the coefficient of base-vector.
1344  for (int j = 0; j < base; ++j) {
1345  for (int i = 0; i < detCalc.RowCount(); ++i) {
1346  detCalc(i, j) = (*(args[i]))[j];
1347  }
1348  }
1349  for (int j = base + 1; j < result.Dimension(); ++j) {
1350  for (int i = 0; i < detCalc.RowCount(); ++i) {
1351  detCalc(i, j - 1) = (*(args[i]))[j];
1352  }
1353  }
1354 
1355  T coefficient = T(sign) * detCalc.Determinant();
1356  result(base) = coefficient;
1357  }
1358 
1359  return result;
1360 }
