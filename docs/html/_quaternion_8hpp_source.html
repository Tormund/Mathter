281  template <class U, int Rows, int Columns, eMatrixOrder Order, eMatrixLayout Layout, bool Packed>
282  Matrix<U, Rows, Columns, Order, Layout, Packed> ToMatrix() const {
283  assert(IsNormalized());
284  Matrix<U, Rows, Columns, Order, Layout, Packed> mat;
285  auto elem = [&mat](int i, int j) -> U& {
286  return Order == eMatrixOrder::PRECEDE_VECTOR ? mat(i, j) : mat(j, i);
287  };
288  elem(0, 0) = 1 - 2 * (j*j + k*k); elem(0, 1) = 2 * (i*j - k*s); elem(0, 2) = 2 * (i*k + j*s);
289  elem(1, 0) = 2 * (i*j + k*s); elem(1, 1) = 1 - 2 * (i*i + k*k); elem(1, 2) = 2 * (j*k - i*s);
290  elem(2, 0) = 2 * (i*k - j*s); elem(2, 1) = 2 * (j*k + i*s); elem(2, 2) = 1 - 2 * (i*i + j*j);
291 
292  // Rest
293  for (int j = 0; j < mat.Width(); ++j) {
294  for (int i = (j < 3 ? 3 : 0); i < mat.Height(); ++i) {
295  mat(i, j) = T(j == i);
296  }
297  }
298 
299  return mat;
300  }

302  template <class U, int Rows, int Columns, eMatrixOrder Order, eMatrixLayout Layout, bool Packed>
303  void FromMatrix(const Matrix<U, Rows, Columns, Order, Layout, Packed>& mat) {
304  assert(mat.IsRotationMatrix3D());
305  auto elem = [&mat](int i, int j) -> U {
306  return Order == eMatrixOrder::PRECEDE_VECTOR ? mat(i, j) : mat(j, i);
307  };
308  w = std::sqrt(1 + elem(0, 0) + elem(1, 1) + elem(2, 2)) * T(0.5);
309  T div = T(1) / (T(4) * w);
310  x = (elem(2, 1) - elem(1, 2)) * div;
311  y = (elem(0, 2) - elem(2, 0)) * div;
312  z = (elem(1, 0) - elem(0, 1)) * div;
313  }

315  template <bool SimdAccelerated>
316  inline static Quaternion Product(const Quaternion& lhs, const Quaternion& rhs) {
317  Quaternion ret;
318  ret.w = lhs.s*rhs.s - lhs.x*rhs.x - lhs.y*rhs.y - lhs.z*rhs.z;
319  ret.x = lhs.s*rhs.x + lhs.x*rhs.s + lhs.y*rhs.z - lhs.z*rhs.y;
320  ret.y = lhs.s*rhs.y - lhs.x*rhs.z + lhs.y*rhs.s + lhs.z*rhs.x;
321  ret.z = lhs.s*rhs.z + lhs.x*rhs.y - lhs.y*rhs.x + lhs.z*rhs.s;
322  return ret;
323  }

325  template <>
326  inline static Quaternion Product<true>(const Quaternion& lhs, const Quaternion& rhs) {
327  Quaternion ret;
328  using SimdT = Simd<T, 4>;
329 
330  SimdT dabc = lhs.vec.simd;
331  SimdT wxyz = rhs.vec.simd;
332  SimdT alternate;
333  alternate.v[0] = -1;
334  alternate.v[1] = 1;
335  alternate.v[2] = -1;
336  alternate.v[3] = 1;
337 
338  // [ 3, 2, 1, 0 ]
339  // [ 0, 3, 2, 1 ]
340  SimdT t0 = SimdT::shuffle<0,0,0,0>(dabc);
341  SimdT t1 = SimdT::shuffle<3,0,1,2>(wxyz);
342 
343  SimdT t2 = SimdT::shuffle<1,1,1,1>(dabc);
344  SimdT t3 = SimdT::shuffle<2,1,0,3>(wxyz);
345 
346  SimdT t4 = SimdT::shuffle<2,2,2,2>(dabc);
347  SimdT t5 = SimdT::shuffle<3,1,0,2>(wxyz);
348 
349  SimdT m0 = SimdT::mul(t0, t1);
350  SimdT m1 = SimdT::mul(t2, t3);
351  SimdT m2 = SimdT::mul(t4, t5);
352 
353  SimdT t6 = SimdT::shuffle<3,3,3,3>(dabc);
354  SimdT t7 = SimdT::shuffle<0,3,1,2>(wxyz);
355 
356  SimdT m3 = SimdT::mul(t6, t7);
357 
358  SimdT e = SimdT::add(m0, SimdT::mul(alternate, m1));
359  e = SimdT::shuffle<1, 3, 0, 2>(e);
360  e = SimdT::add(e, SimdT::mul(alternate, m2));
361  e = SimdT::shuffle<2, 0, 1, 3>(e);
362  e = SimdT::add(e, SimdT::mul(alternate, m3));
363  e = SimdT::shuffle<3, 1, 0, 2>(e);
364 
365  ret.vec.simd = e;
366  return ret;
367  }
